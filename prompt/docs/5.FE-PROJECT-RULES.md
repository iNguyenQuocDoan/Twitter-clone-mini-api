## 2. Server Components vs Client Components

This is the most critical decision in Next.js App Router. Default to **Server Components**; add `'use client'` only when necessary.

### Decision Matrix

| Needs | Component Type | Directive |
|-------|---------------|-----------|
| Fetch data from API (no interactivity) | Server Component | none (default) |
| Use `useState`, `useEffect`, `useReducer` | Client Component | `'use client'` |
| Use TanStack Query / Zustand | Client Component | `'use client'` |
| Handle click events, forms, animations | Client Component | `'use client'` |
| Access cookies/headers on server | Server Component | none |
| Use browser APIs (`window`, `localStorage`) | Client Component | `'use client'` |
| Render static/SEO content | Server Component | none |

### Patterns

```tsx
// ✅ Server Component — fetch data, no interactivity
// app/(public)/products/page.tsx
export default async function ProductListPage({
  searchParams,
}: {
  searchParams: { page?: string; category_id?: string };
}) {
  const data = await serverFetch<PaginatedResponse<Product>>(
    `/products?page=${searchParams.page ?? 1}&category_id=${searchParams.category_id ?? ''}`,
  );

  return <ProductList initialData={data} />;
}

// ✅ Client Component — interactive, uses hooks
// features/product/components/ProductList.tsx
'use client';

import { useProducts } from '../hooks/useProducts';

export const ProductList = ({
  initialData,
}: {
  initialData: PaginatedResponse<Product>;
}) => {
  const { data } = useProducts({ initialData }); // hydrate from server data
  return <div>{data?.data.map((p) => <ProductCard key={p.id} product={p} />)}</div>;
};

// ❌ DON'T: Add 'use client' to a page that only renders static content
'use client'; // WRONG — forces entire subtree to client bundle
export default function ProductListPage() { ... }
```

## 5. Code Patterns

### Data Fetching — Server Components (RSC)

```ts
// shared/lib/fetch.ts — server-side fetch wrapper
import { cookies } from 'next/headers';

export async function serverFetch<T>(path: string, options?: RequestInit): Promise<T> {
  const cookieStore = await cookies();
  const token = cookieStore.get('accessToken')?.value;

  const res = await fetch(`${process.env.API_BASE_URL}${path}`, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
      ...options?.headers,
    },
    next: { revalidate: 60 }, // ISR: revalidate every 60 seconds
  });

  if (!res.ok) throw new Error(`Fetch failed: ${res.status}`);
  return res.json();
}
```

> **When to use `serverFetch`:** Pages that benefit from SSR/ISR (product listing, product detail, order history). Data is fetched on the server — no loading spinner, better SEO, faster LCP.

### Data Fetching — Client Components (TanStack Query)

```tsx
// features/product/services/product.service.ts — client-side
import { apiClient } from '@/shared/lib/axios';
import type { PaginatedResponse, ApiResponse } from '@/shared/types';
import type { Product, ProductParams } from '../types/product.types';

export const productService = {
  getAll: (params: ProductParams) =>
    apiClient.get<PaginatedResponse<Product>>('/products', { params }),
  getBySlug: (slug: string) =>
    apiClient.get<ApiResponse<Product>>(`/products/${slug}`),
};

// features/product/hooks/useProducts.ts
'use client';
export const useProducts = (params: ProductParams & { initialData?: PaginatedResponse<Product> }) => {
  const { initialData, ...queryParams } = params;
  return useQuery({
    queryKey: ['products', queryParams],
    queryFn: () => productService.getAll(queryParams),
    initialData,              // hydrate from RSC server fetch
    staleTime: 60 * 1000,    // 1 minute
  });
};
```

### Route Protection — Middleware

```ts
// middleware.ts
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

const PROTECTED_PATHS = ['/cart', '/checkout', '/orders', '/profile'];
const ADMIN_PATHS = ['/admin'];

export function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl;
  const accessToken = request.cookies.get('accessToken')?.value;

  const isProtected = PROTECTED_PATHS.some((p) => pathname.startsWith(p));
  const isAdmin = ADMIN_PATHS.some((p) => pathname.startsWith(p));

  if ((isProtected || isAdmin) && !accessToken) {
    return NextResponse.redirect(new URL('/login', request.url));
  }

  // Role check — decode JWT to verify role
  if (isAdmin) {
    const role = getTokenRole(accessToken!); // decode without verify (trusted server)
    if (role !== 'admin') {
      return NextResponse.redirect(new URL('/', request.url));
    }
  }

  return NextResponse.next();
}

export const config = {
  matcher: ['/cart/:path*', '/checkout/:path*', '/orders/:path*', '/profile/:path*', '/admin/:path*'],
};
```

### Form Handling

```tsx
'use client';

const schema = z.object({
  email: z.string().email('Invalid email'),
  password: z.string().min(8, 'Minimum 8 characters'),
});

type FormValues = z.infer<typeof schema>;

export const LoginForm = () => {
  const { register, handleSubmit, formState: { errors, isSubmitting } } = useForm<FormValues>({
    resolver: zodResolver(schema),
  });
  const loginMutation = useLogin();

  return (
    <form onSubmit={handleSubmit((data) => loginMutation.mutate(data))}>
      <Input {...register('email')} error={errors.email?.message} />
      <Input type="password" {...register('password')} error={errors.password?.message} />
      <Button type="submit" loading={isSubmitting || loginMutation.isPending}>
        Login
      </Button>
    </form>
  );
};
```

### Authentication — Token Strategy

Access token stored in **httpOnly cookie** (set by NestJS backend). Next.js middleware reads this cookie server-side for route protection. Zustand stores user info (not token) for client-side UI.

```tsx
// shared/lib/axios.ts — client-side only
'use client';

const apiClient = axios.create({ baseURL: process.env.NEXT_PUBLIC_API_BASE_URL });

// Attach token from cookie (readable by JS if not httpOnly — use separate short-lived cookie)
// OR: rely on cookies being sent automatically with same-domain requests
apiClient.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response?.status === 401 && !error.config._retry) {
      error.config._retry = true;
      try {
        await apiClient.post('/auth/refresh'); // refresh token sent via httpOnly cookie
        return apiClient(error.config);
      } catch {
        useAuthStore.getState().logout();
        window.location.href = '/login';
      }
    }
    return Promise.reject(error);
  },
);

export { apiClient };
```

## 6. Anti-Patterns (MUST NOT)

| ❌ DON'T | ✅ DO | Why |
|----------|-------|-----|
| `'use client'` on every component | Default to RSC, add only when needed | Unnecessary client bundle bloat |
| `useEffect` + `fetch` in Client Component | `serverFetch` in RSC or TanStack Query | RSC: zero JS cost; TanStack: caching |
| `localStorage` for access token | httpOnly cookie (set by server) | XSS mitigation |
| `process.env.SECRET` in Client Component | Only in Server Components or API routes | Exposed to browser |
| Import feature internals | Use barrel `index.ts` | Feature contract enforcement |
| Direct API calls in components | Service + TanStack Query hook | Caching, deduplication |
| Store server data in Zustand | TanStack Query cache | Avoids stale data |
| `any` type or `ts-ignore` | Explicit types | TypeScript strict mode |
| Prop drilling > 2 levels | Context or Zustand | Maintainability |
| `import from 'react-router-dom'` | Use Next.js `useRouter`, `usePathname`, `useSearchParams` | Next.js has its own router |

## 7. Component Rules

```tsx
// Server Component — no directive, async function
// app/(public)/products/[slug]/page.tsx
export default async function ProductDetailPage({ params }: { params: { slug: string } }) {
  const product = await serverFetch<ApiResponse<Product>>(`/products/${params.slug}`);
  return <ProductDetailClient product={product.data} />;
}

// Client Component — interactive parts only
// features/product/components/ProductDetailClient.tsx
'use client';

interface Props { product: Product }

export const ProductDetailClient = ({ product }: Props) => {
  const [selectedVariant, setSelectedVariant] = useState(product.variants[0]);
  const { mutate: addToCart } = useAddToCart();

  return (
    <div>
      <ProductImageGallery images={product.images} />
      <VariantSelector
        variants={product.variants}
        selected={selectedVariant}
        onSelect={setSelectedVariant}
      />
      <Button onClick={() => addToCart({ variantId: selectedVariant.id, quantity: 1 })}>
        Add to Cart
      </Button>
    </div>
  );
};
```

**Component checklist:**
- [ ] Default to Server Component — add `'use client'` only when needed
- [ ] Props typed explicitly — no `any`
- [ ] Loading state: use `loading.tsx` (RSC) or skeleton component (Client)
- [ ] Error state: use `error.tsx` (RSC) or inline error (Client)
- [ ] Max 200 lines — extract sub-components if exceeded
- [ ] No direct API calls in component body
